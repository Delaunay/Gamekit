UENUM(BlueprintType)
enum class EGKRelativePosition : uint8
{
    None  = 0,
    Top   = 1, // 0001
    Bot   = 2, // 0010
    Left  = 4, // 0100
    Right = 8, // 1000

    TopLeft  = Top | Left,  // 0101 i.e  5
    BotLeft  = Bot | Left,  // 0110 i.e  6
    TopRight = Top | Right, // 1001 i.e  9
    BotRight = Bot | Right, // 1010 i.e 10
};

};
